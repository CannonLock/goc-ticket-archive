    I'm sorry what exactly do you want GLOW to do here?  Our proxy should be good until 2016 afaik...

Here is how we keep up our proxies on glidein2.chtc.wisc.edu
# renew proxy every 3 days.  Max time for glow-voms proxy is 3 days so renew both
0 0 */3 * * /etc/condor/certs/renew_proxies.sh

[root@glidein2 ~]# cat /etc/condor/certs/renew_proxies.sh
#!/bin/sh

cd /etc/condor/certs

# renew frontend proxy
voms-proxy-init -q \
-cert glidein.cert -key glidein.key \
-out glidein.proxy -valid 8760:00

# renew pilot proxy
voms-proxy-init -q \
-cert pilot.cert -key pilot.key \
-out pilot.proxy -voms GLOW:/Role=htpc -valid 8760:00

chown frontend: *.proxy
