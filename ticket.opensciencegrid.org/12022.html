#!/bin/bash

RANDOM_MIN=100
RANDOM_MAX=1000
RANDOM_NUMBER=$(( $RANDOM_MIN+(`od -An -N2 -i /dev/random`
)%($RANDOM_MAX-$RANDOM_MIN+1) ))

RANDOM_SLEEP_MIN=1
RANDOM_SLEEP_MAX=15
RANDOM_SLEEP_NUMBER=$(( $RANDOM_SLEEP_MIN+(`od -An -N2 -i /dev/random`
)%($RANDOM_SLEEP_MAX-$RANDOM_SLEEP_MIN+1) ))

RANDOM_TEXT_MIN=100
RANDOM_TEXT_MAX=2000
RANDOM_TEXT_NUMBER=$(( $RANDOM_TEXT_MIN+(`od -An -N2 -i /dev/random`
)%($RANDOM_TEXT_MAX-$RANDOM_TEXT_MIN+1) ))

# print information
echo "Random number = $RANDOM_NUMBER"
echo "Random text number = $RANDOM_TEXT_NUMBER"

LOOP_IDX=1
LOOP_TEXT_IDX=1

while [ $LOOP_IDX -le $RANDOM_NUMBER ]
do
while [ $LOOP_TEXT_IDX -le $RANDOM_TEXT_NUMBER ]
do
LOOP_TEXT_IDX=`expr $LOOP_TEXT_IDX + 1`
done
#echo "$LOOP_IDX `date`"
LOOP_IDX=`expr $LOOP_IDX + 1`
done
echo "Sleep $RANDOM_SLEEP_NUMBER `date`"
sleep $RANDOM_SLEEP_NUMBER
echo "Slept $RANDOM_SLEEP_NUMBER `date`"
