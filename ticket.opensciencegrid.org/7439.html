    Sep 8, 2009 09:34 PM UTC by Steven Clark 
    Here is a typical example set of scripts.

#!/bin/sh
#
CURDIR=`pwd`

SCRIPT=`basename $0`
SITE_SCRATCH_DIR=$3
TARBALL=${SITE_SCRATCH_DIR}/$1
WORKINGDIR=${SITE_SCRATCH_DIR}/$2

if [ ! -d ${SITE_SCRATCH_DIR} ] ; then
echo "Jobs directory ${SITE_SCRATCH_DIR} is missing"
exit 13
fi

if [ "${USER}" = "" ] ; then
if [ "${LOGNAME}" != "" ] ; then
export USER=${LOGNAME}
else
export USER="nanoHUB"
fi
fi

APP_PROGRAM=./probegridsite.sh
if [ ! -x ${APP_PROGRAM} ] ; then
if [ "${SITE_APPLICATION_ROOT}" != "" ] ; then
if [ -x ${SITE_APPLICATION_ROOT}/${APP_PROGRAM} ] ; then
APP_PROGRAM=${SITE_APPLICATION_ROOT}/${APP_PROGRAM}
fi
fi
fi
STD_INPUT=/dev/null

mkdir -p ${WORKINGDIR}

    chmod 700 ${WORKINGDIR}
tar xzm -C ${WORKINGDIR} -f ${SITE_SCRATCH_DIR}/00517397_input.tar.gz
#
cd ${WORKINGDIR}
sleep 2
touch .__fileTimeMarker
sleep 1

TIMEPATH=
for timePath in ${HOME}/bin/time /usr/bin/time /usr/local/bin/time
/apps/submit/bin/time ; do
if [ -x $timePath ] ; then
TIMEPATH=$timePath
break
fi
done

date +"%s" > .__timestamp_start.00517397
#
# create output file now - if job times out the
# file is available for transfer by condor
#
tar -zcf ${TARBALL} ./.__timestamp_start.00517397
#
export PATH=${PATH}:${WORKINGDIR}

${TIMEPATH} -p -o .__time_results.00517397 ${APP_PROGRAM}  < ${STD_INPUT}

date +"%s" > .__timestamp_finish.00517397
#
# tar: Cannot update compressed archives
# so recreate output file for transfer by condor
#
tar -zcf ${TARBALL} `find . -newer .__fileTimeMarker -not -name . -print`

cd ${CURDIR}
rm -rf ${WORKINGDIR}

probegridsite.sh:

#!/bin/sh
#
exitStatus=0

free -m
printenv | sort

pwd
ls -lsa

if [ "${SITE_EXPECT_HOME_DIR}" = "YES" ] ; then
if [ "${HOME}" != "" ] ; then
cd ${HOME}
exitStatus=$?

if [ $exitStatus -eq 0 ] ; then
du -sh
else
exitStatus=6
fi
fi
fi

exit $exitStatus
