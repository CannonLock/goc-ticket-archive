Malina,

bestman gateway is in VDT test cache currently.
for gateway installation, you need -enable-gateway-mode, and you don't
need storage path info.

# The those two ports, you can use any other open ports
# Java 1.6 is required now.

--with-srm-home=/share/apps/bestman-2.2.0.11/bestman \
--with-srm-owner=daemon \
--with-cacert-path=/share/apps/bestman-2.2.0.11/globus/TRUSTED_CA \
--with-certfile-path=/etc/grid-security/http/httpcert.pem \
--with-keyfile-path=/etc/grid-security/http/httpkey.pem \

--with-eventlog-path=/share/apps/bestman-2.2.0.11/vdt-app-data/bestman/logs
\

--with-cachelog-path=/share/apps/bestman-2.2.0.11/vdt-app-data/bestman/logs
\
--with-http-port=7070 \                # <------necessary for CMS utils?
YES
--with-https-port=8443 \               # <------necessary for CMS utils?
YES
# The those two above ports, you can use any other open ports.
--with-globus-location=/share/apps/bestman-2.2.0.11/globus \
--with-java-home=/share/apps/bestman-2.2.0.11/jdk1.5 \
# Java 1.6 is required now.
--with-globus-tcp-port-range=20000,25000 \  # <------ necessaryfor CMS
utils
--with-globus-tcp-source-range=20000,25000  # <------ necessary for CMS
utils
