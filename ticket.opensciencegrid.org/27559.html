    Hi Gracie Jane,

At ANL we have an SRM/gridftp server that uses disks served from a NFS file server.

Here is the script that I run on the file server monthly in a cron file -

[root@atlasfs02 ~]# cat /etc/cron.monthly/posix_dump.sh
#!/bin/sh
set -e

time_stamp=`date -u +'%Y%m%d'`
prefix='/grid/atlas/dq2/'
spacetokens='atlasscratchdisk,atlasdatadisk,atlasuserdisk'

dprefix="$prefix"
dspacetokens="$spacetokens"

usage(){
(
printf "Usage: $0 [--prefix <prefix>] "
printf "[--spacetokens <spctkn1[,spctkn2,...]>]\n"
printf '\nOptions:\n'
printf "  --prefix|-p      Prefix of the files in the local filesystem (full path)\n"
printf "                   [default: \"$dprefix\"]\n"
printf "  --spacetokens|-s Spacetokens to include in the dumps\n"
printf "                   [default: \"$dspacetokens\"]\n"
) 1>&2
}

while [ $# -gt 0 ]; do
case "$1" in
--prefix|-p)
shift
prefix="$1"
;;

    --spacetokens|-s)
shift
spacetokens="$1"
;;
--help|-h)
usage
exit 0
;;
*)
usage
exit 1
esac
shift
done

prefix=/$(echo "$prefix" | sed -r 's@/+$@@' | sed -r 's@^/+@@')
discard=$(expr $(echo "$prefix" | tr -cd / | wc -m) + 3)

for spacetoken in $(echo $spacetokens | tr ',' ' ')
do
# convert lower case to upper case
uc_spacetoken=`echo $spacetoken | tr '[:lower:]' '[:upper:]' `
find ${prefix}/${uc_spacetoken} -type f | cut -d/ -f${discard}- | sed -r 's@^rucio/@@' > /tmp/${spacetoken}-dump.out
mv  -f /tmp/${spacetoken}-dump.out ${prefix}/${uc_spacetoken}/dumps/dump_${time_stamp}
chown usatlas2:atlas ${prefix}/${uc_spacetoken}/dumps/dump_${time_stamp}
done
-----
These are the mount points on file server
/dev/mapper/vg_extdisk1-lvdataDisk
2.8T  645G  2.1T  24% /grid/atlas/dq2/ATLASDATADISK
/dev/mapper/vg_intdisk-lvuserdisk
5.4T  2.8T  2.5T  54% /grid/atlas/dq2/ATLASUSERDISK
/dev/mapper/vg_extdisk1-lvscratchDisk
3.2T  1.6T  1.6T  50% /grid/atlas/dq2/ATLASSCRATCHDISK

Cheers,

Doug
